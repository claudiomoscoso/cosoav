use cosoav;

drop table if exists tDataBrute;
drop table if exists tIn;
drop table if exists tOut;
drop table if exists tReportStatus;
drop table if exists tReport;
drop table if exists tTimeConfig;
drop table if exists tPlainType;
drop table if exists tAtention;

create table tReportStatus(
	cId INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	cDate Datetime NOT NULL,
	cProcessTime long
	) engine=innoDB;

create table tDataBrute(
	cId INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	cHTML Longtext NOT NULL,
	cXML Longtext,
	cProcessTime Datetime,
	cStatus ENUM('NEW','PROC', 'AS_XML', 'PROC_XML', 'DONE', 'ERROR', 'ERROR_XML', 'BUILD_REPO', 'OK', 'ERROR_REPO') DEFAULT 'NEW',
	cDate Date
	) engine=innoDB;

create table tIn(
	cID	INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	cDataBrute	INT NOT NULL,
	MAT	CHAR(6),
	ST	CHAR(3),
	CRD1	CHAR(6),
	VUELO	CHAR(5),
	ORI	CHAR(3),
	ITI	CHAR(5),
	RECNF	CHAR(5),
	OBS	VARCHAR(250),
	TA	CHAR(5),
	cDate DateTime
	) engine=innoDB;

create table tOut(
	cID	INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	cDataBrute	INT NOT NULL,
	MAT	CHAR(6),	
	ST	CHAR(3),
	PTA	CHAR(3),
	CRD1	CHAR(3),
	CRD2	CHAR(3),
	VUELO	CHAR(4),
	DES	CHAR(3),
	ITI	CHAR(5),	
	CNF	CHAR(5),
	ROC	CHAR(6),
	ATR	VARCHAR(250),
	OBS	VARCHAR(250),
	cDate DateTime,
	cUsed	BOOLEAN DEFAULT '0'
	) engine=innoDB;

create table tReport(
	cID	INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	cDataBrute	INT NOT NULL,
	MAT	CHAR(6),
	ST	CHAR(3),
	VUELO	CHAR(4),
	cLlegada	DATETIME,
	cSalida	DATETIME,
	cTransito	INT,
	cAtention	INT,
	cStatus	ENUM('NEW', 'DONE') DEFAULT 'NEW'
) engine=innoDB;

create table tAtention(
	cID	INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	cName	char(6)	
) engine=innoDB;

create table tPlainType(
	cID	INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	MAT	CHAR(6),
	cTypePlain char(3)	
) engine=innoDB;

create table tTimeConfig(
	cID	INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
	cTypePlain char(3),
	cMins	INTEGER	
) engine=innoDB;
